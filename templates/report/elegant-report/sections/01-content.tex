\section{Introduction}

Common environments are available: \texttt{definition}, \texttt{theorem}, \texttt{lemma},
\texttt{remark}, \texttt{corollary}, \texttt{problem}, \texttt{claim}.

\begin{definition}
    A metric space $(X, d)$ is \emph{complete} if every Cauchy sequence converges in $X$.
\end{definition}

\begin{theorem}
    Every compact metric space is complete.
\end{theorem}

\begin{remark}
    The converse is not true: $\mathbb{R}$ is complete but not compact.
\end{remark}

\begin{lemma}
    Let $(X,d)$ be compact. For any $\varepsilon>0$, finitely many $\varepsilon$-balls cover $X$.
\end{lemma}

\begin{corollary}
    A closed bounded subset of $\mathbb{R}^n$ is compact.
\end{corollary}

\begin{problem}
    Prove that $[0,1]$ with the standard metric is complete.
\end{problem}
